```cpp
#include <iostream>
#include <vector>
#include <stack>
using namespace std;

// 定义所有可能的运算符（5种）
const char ops_list[5] = {'+', '-', '*', '/', '%'};
int K; // 目标结果值，通过输入获取

// 检查给定的运算符位置是否构成合法的后缀表达式
bool is_valid(const vector<int>& pos) {
    vector<bool> is_op(7, false); // 标记每个位置是否为运算符
    for (int p : pos) is_op[p] = true;

    int stack_size = 0; // 模拟栈的大小
    for (int i = 0; i < 7; i++) {
        if (is_op[i]) { // 当前位置是运算符
            if (stack_size < 2) return false; // 栈中元素不足两个，无法运算
            stack_size--; // 消耗两个操作数，生成一个结果
        } else { // 当前位置是数字
            stack_size++; // 压入栈中
        }
    }
    return stack_size == 1; // 最终栈中必须剩下一个结果
}

int main() {
    cin >> K; // 输入目标结果K

    vector<vector<int>> valid_positions; // 存储所有合法的运算符位置组合

    // 预处理阶段：生成所有可能的3个运算符的位置组合，并筛选合法的组合
    for (int a = 0; a < 7; a++) { // 第一个运算符的位置
        for (int b = a + 1; b < 7; b++) { // 第二个运算符的位置（必须在a之后）
            for (int c = b + 1; c < 7; c++) { // 第三个运算符的位置（必须在b之后）
                vector<int> pos = {a, b, c}; // 当前位置组合
                if (is_valid(pos)) { // 验证是否合法
                    valid_positions.push_back(pos); // 保存合法组合
                }
            }
        }
    }

    int cnt = 0; // 统计符合条件的表达式数量

    // 遍历所有合法的运算符位置组合
    for (const auto& valid_pos : valid_positions) {
        vector<bool> is_op(7, false); // 标记每个位置是否为运算符
        for (int p : valid_pos) is_op[p] = true;

        // 预计算数字位置的索引映射：
        // digits_positions存储所有数字的位置顺序
        // digit_index_map[i]表示位置i在digits数组中的索引
        vector<int> digits_positions;
        vector<int> digit_index_map(7, -1);
        for (int i = 0; i < 7; i++) {
            if (!is_op[i]) { // 如果是数字位置
                digits_positions.push_back(i); // 记录位置
                digit_index_map[i] = digits_positions.size() - 1; // 映射到digits数组的索引
            }
        }

        // 预计算运算符索引：
        // op_indices[i]表示位置i对应的运算符在ops数组中的索引（-1表示非运算符位置）
        vector<int> op_indices(7, -1);
        for (int k = 0; k < 3; k++) {
            op_indices[valid_pos[k]] = k; // 记录运算符的位置对应关系
        }

        // 遍历所有可能的运算符组合（5^3=125种）
        for (int op_num = 0; op_num < 125; op_num++) {
            // 将op_num分解为三个运算符的选择（每位5进制）
            int op0 = op_num / 25; // 第一个运算符的索引
            int op1 = (op_num % 25) / 5; // 第二个运算符的索引
            int op2 = op_num % 5; // 第三个运算符的索引
            vector<char> ops = {ops_list[op0], ops_list[op1], ops_list[op2]}; // 当前运算符组合

            // 遍历所有可能的4位数字组合（0-9999）
            for (int num = 0; num < 10000; num++) {
                // 将数字分解为4个独立的数字（d0-d3）
                int d0 = num / 1000; // 千位
                int d1 = (num / 100) % 10; // 百位
                int d2 = (num / 10) % 10; // 十位
                int d3 = num % 10; // 个位
                vector<int> digits = {d0, d1, d2, d3}; // 当前数字组合

                // 模拟后缀表达式计算过程
                bool valid_expr = true; // 表达式是否合法
                int stack[7]; // 模拟栈（最大深度7）
                int sp = 0; // 栈指针

                for (int i = 0; i < 7 && valid_expr; i++) { // 遍历每个字符位置
                    if (op_indices[i] != -1) { // 当前位置是运算符
                        int op_k = op_indices[i]; // 当前运算符在ops数组中的索引
                        char current_op = ops[op_k]; // 获取当前运算符

                        // 检查栈是否足够执行运算
                        if (sp < 2) {
                            valid_expr = false; continue;
                        }

                        // 取出两个操作数（栈顶元素在前）
                        int b = stack[sp - 1];
                        int a = stack[sp - 2];
                        sp -= 2; // 弹出两个操作数

                        // 检查除零或模零错误
                        if ((current_op == '/' || current_op == '%') && b == 0) {
                            valid_expr = false; continue;
                        }

                        // 执行运算并压入结果
                        int res = 0; // 初始化结果（避免未初始化警告）
                        switch (current_op) {
                            case '+': res = a + b; break;
                            case '-': res = a - b; break;
                            case '*': res = a * b; break;
                            case '/': res = a / b; break;
                            case '%': res = a % b; break;
                        }
                        stack[sp++] = res;

                    } else { // 当前位置是数字
                        int digit_pos = digit_index_map[i]; // 获取该数字在digits数组中的索引
                        stack[sp++] = digits[digit_pos]; // 将数字压入栈
                    }
                }

                // 检查最终结果是否符合条件
                if (valid_expr && sp == 1 && stack[0] == K) {
                    cnt++; // 统计符合条件的表达式
                }
            }
        }
    }

    cout << cnt << endl; // 输出结果
    return 0;
}
```

```cpp
#include <iostream>
#include <vector>
#include <stack>
#include <cassert>
using namespace std;

const char ops_list[5] = {'+', '-', '*', '/', '%'};
int K;

struct Step {
    int left; // 数字索引或中间结果的编号（-1, -2等）
    int right;
};

struct ExpressionSteps {
    Step step0;
    Step step1;
    Step step2;
};

struct PositionData {
    vector<int> pos;
    ExpressionSteps steps;
};

bool is_valid(const vector<int>& pos) {
    vector<bool> is_op(7, false);
    for (int p : pos) is_op[p] = true;
    int stack_size = 0;
    for (int i = 0; i < 7; i++) {
        if (is_op[i]) {
            if (stack_size < 2) return false;
            stack_size--;
        } else {
            stack_size++;
        }
    }
    return stack_size == 1;
}

// 预处理每个位置组合的表达式结构
vector<PositionData> preprocess_positions() {
    vector<PositionData> valid_positions;
    for (int a = 0; a < 7; a++) {
        for (int b = a + 1; b < 7; b++) {
            for (int c = b + 1; c < 7; c++) {
                vector<int> pos = {a, b, c};
                if (is_valid(pos)) {
                    // 计算表达式步骤
                    ExpressionSteps steps;
                    vector<bool> is_op(7, false);
                    for (int p : pos) is_op[p] = true;
                    
                    // 预计算数字位置映射
                    vector<int> digits_positions;
                    vector<int> digit_index_map(7, -1);
                    for (int i = 0; i < 7; i++) {
                        if (!is_op[i]) {
                            digits_positions.push_back(i);
                            digit_index_map[i] = digits_positions.size() - 1;
                        }
                    }
                    
                    // 模拟栈操作记录步骤
                    stack<int> s;
                    int current_intermediate = 0;
                    vector<Step> steps_list;
                    for (int i = 0; i < 7; i++) {
                        if (is_op[i]) {
                            int right = s.top(); s.pop();
                            int left = s.top(); s.pop();
                            steps_list.push_back({left, right});
                            s.push(- (current_intermediate + 1));
                            current_intermediate++;
                        } else {
                            int digit_pos = digit_index_map[i];
                            s.push(digit_pos);
                        }
                    }
                    
                    assert(steps_list.size() == 3);
                    steps.step0 = steps_list[0];
                    steps.step1 = steps_list[1];
                    steps.step2 = steps_list[2];
                    
                    valid_positions.push_back({pos, steps});
                }
            }
        }
    }
    return valid_positions;
}

int main() {
    cin >> K;
    vector<PositionData> valid_positions = preprocess_positions();
    int cnt = 0;

    for (const auto& pos_data : valid_positions) {
        const auto& steps = pos_data.steps;
        
        // 遍历所有运算符组合（5^3=125种）
        for (int op_num = 0; op_num < 125; op_num++) {
            int op0 = op_num / 25;
            int op1 = (op_num % 25) / 5;
            int op2 = op_num % 5;
            char ops[3] = {ops_list[op0], ops_list[op1], ops_list[op2]};
            
            // 遍历所有数字组合（10^4种）
            for (int num = 0; num < 10000; num++) {
                vector<int> digits = {
                    num / 1000,
                    (num / 100) % 10,
                    (num / 10) % 10,
                    num % 10
                };
                
                // 计算步骤结果
                bool valid = true;
                int res0 = 0, res1 = 0, final_res = 0;
                
                // 步骤0：第一个运算符
                int l0 = steps.step0.left;
                int r0 = steps.step0.right;
                int val_l0 = digits[l0];
                int val_r0 = digits[r0];
                
                if ((ops[0] == '/' || ops[0] == '%') && val_r0 == 0) {
                    valid = false;
                } else {
                    switch (ops[0]) {
                        case '+': res0 = val_l0 + val_r0; break;
                        case '-': res0 = val_l0 - val_r0; break;
                        case '*': res0 = val_l0 * val_r0; break;
                        case '/': res0 = val_l0 / val_r0; break;
                        case '%': res0 = val_l0 % val_r0; break;
                    }
                }
                
                if (!valid) continue;
                
                // 步骤1：第二个运算符
                int l1 = steps.step1.left;
                int r1 = steps.step1.right;
                int val_l1, val_r1;
                
                if (l1 >= 0) val_l1 = digits[l1];
                else val_l1 = res0;
                
                if (r1 >= 0) val_r1 = digits[r1];
                else val_r1 = res0;
                
                if ((ops[1] == '/' || ops[1] == '%') && val_r1 == 0) {
                    valid = false;
                } else {
                    switch (ops[1]) {
                        case '+': res1 = val_l1 + val_r1; break;

                }

                    cnt++;
                }
            }
```
